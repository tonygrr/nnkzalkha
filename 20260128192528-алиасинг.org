Ниже рассмотрен пример моделирования двух синусоид, отсчеты которых неотличимы друг от друга.

#+property: header-args :exports both :results file replace
#+begin_src python :results file :outputs both :var fname="attachments/aliasing_plot.png"
  import matplotlib
  matplotlib.use('Agg') # Убираем ошибку
  import matplotlib.pyplot as plt
  import numpy as np


  # Параметры
  fs = 1000  # Частота дискретизации 1000 Гц
  f1 = 100   # Основная частота
  f2 = 1100  # "Зеркальная" частота (f1 + fs)

  t_continuous = np.linspace(0, 0.02, 1000)  # Непрерывное время для графиков
  t_discrete = np.arange(0, 0.02, 1/fs)      # Моменты взятия отсчетов

  # Сигналы
  y1_cont = np.cos(2 * np.pi * f1 * t_continuous)
  y2_cont = np.cos(2 * np.pi * f2 * t_continuous)

  y1_disc = np.cos(2 * np.pi * f1 * t_discrete)
  y2_disc = np.cos(2 * np.pi * f2 * t_discrete)

  # Визуализация
  plt.figure(figsize=(10, 5))
  plt.plot(t_continuous, y1_cont, label=f'Сигнал {f1} Гц', alpha=0.6, color='blue')
  plt.plot(t_continuous, y2_cont, label=f'Сигнал {f2} Гц (высокая частота)', alpha=0.4, color='red', linestyle='--')

  # Точки дискретизации (отсчеты)
  plt.scatter(t_discrete, y1_disc, color='blue', label='Отсчеты (100 Гц)')
  plt.scatter(t_discrete, y2_disc, color='red', marker='x', s=100, label='Отсчеты (1100 Гц)')

  plt.title(f'Алиасинг: частоты {f1} Гц и {f2} Гц при fs={fs} Гц')
  plt.xlabel('Время (с)')
  plt.ylabel('Амплитуда')
  plt.legend(loc='upper right')
  plt.grid(True)

  plt.savefig(fname)
  return fname
#+end_src
